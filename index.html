import yfinance as yf import pandas_ta as ta import time def calculate_probability(symbol): try: # पिछले 2 दिनों का 5-मिनट डेटा (15 मिनट के ट्रेंड के लिए बेस्ट) df = yf.download(symbol, period="2d", interval="5m", progress=False) if df.empty: return "Data Error", 0 # इंडिकेटर्स जोड़ें df['RSI'] = ta.rsi(df['Close'], length=14) df['EMA_9'] = ta.ema(df['Close'], length=9) df['EMA_21'] = ta.ema(df['Close'], length=21) last_row = df.iloc[-1] rsi = last_row['RSI'] ema9 = last_row['EMA_9'] ema21 = last_row['EMA_21'] close = last_row['Close'] prob = 50 # न्यूट्रल # UP Trend के लिए पॉइंट्स (+): if rsi < 35: prob += 25 # Oversold (बाउंस आने के चांस) if ema9 > ema21: prob += 15 # Bullish Crossover if close > ema9: prob += 10 # Price Strength # DOWN Trend के लिए पॉइंट्स (-): if rsi > 65: prob -= 25 # Overbought (गिरने के चांस) if ema9 < ema21: prob -= 15 # Bearish Crossover if close < ema9: prob -= 10 # Price Weakness # सिग्नल तय करें if prob >= 70: signal = "STRONG BUY 🟢" elif prob > 55: signal = "BUY 📈" elif prob <= 30: signal = "STRONG SELL 🔴" elif prob < 45: signal = "SELL 📉" else: signal = "WAITING... 🟡" return signal, prob except Exception as e: return f"Error: {e}", 50 # लगातार चलाने के लिए लूप while True: nifty_signal, nifty_prob = calculate_probability("^NSEI") print(f"--- NIFTY 50 LIVE ---") print(f"Signal: {nifty_signal}") print(f"Probability: {nifty_prob}%") print("-" * 20) # मार्केट की चाल धीमी होती है, इसलिए हर 5 मिनट में अपडेट करें time.sleep(300)
