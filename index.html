async function refreshMarketData() { try { const response = await fetch(CSV_URL); const text = await response.text(); const rows = text.split('\n'); // Nifty Data (Row 1) const nPrice = parseFloat(rows[0].split(',')[0]); const nEMA = parseFloat(rows[0].split(',')[1]); // Sensex Data (Row 2) const sPrice = parseFloat(rows[1].split(',')[0]); const sEMA = parseFloat(rows[1].split(',')[1]); if (nPrice) { // Nifty Update with Trend const nTrend = nPrice > nEMA ? "Bullish 📈" : "Bearish 📉"; document.getElementById('nifty').innerHTML = `₹ ${nPrice.toLocaleString('en-IN')}
Trend: ${nTrend}`; // Sensex Update with Trend const sTrend = sPrice > sEMA ? "Bullish 📈" : "Bearish 📉"; document.getElementById('sensex').innerHTML = `₹ ${sPrice.toLocaleString('en-IN')}
Trend: ${sTrend}`; document.getElementById('update-msg').innerText = "Live: " + new Date().toLocaleTimeString(); } } catch (err) { document.getElementById('update-msg').innerText = "Sync Error!"; } }
